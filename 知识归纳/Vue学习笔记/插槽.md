##### 默认插槽

```html
[Vue Component].vue (Parent)

	...
	<Son_1 :list="son1">
		<!-- 该组件标签的标签体内容，显示在该组件里的<slot></slot>中 -->
		<!-- 该标签体内容是在<Parent>组件里解析完成后，再插入<Son>组件中 -->
		<span>Parent - Son-1</span>
	</Son_1>
	
	<Son_2 :list="son2">
		<img src="001.jpg" />
	</Son_2>
	
	<Son_3 :list="son3">
		<a href="https://www.baidu.com/"></a>
	</Son_3>


	<script>
		...
		data() {
			return {
				son1: ['son1-one', 'son1-two'],
				son2: ['son2-one', 'son2-two'],
				son3: ['son3-one', 'son3-two']
			};
		}
		...
	</script>


	<!-- scoped私有化，只控制当前模块 -->
	<!-- 
		因<Son>标签体内容是在当前模块<Parent>组件里解析完成后，再插入<Son>组件中，所以即使样式私有化，也能控制<Son>标签体内容的样式
		
	 -->
	<style scoped>
		span {...}
		img {...}
		a {...}
	</style>
```

```html
[Vue Component].vue (Son)

	...
	<li>
		<ul v-for="(item, index) in list" :key="index">{{item}}</ul>
		<!-- slot标签体中可有默认内容，若没有传递具体结构（父组件中该组件标签体内容）时显示 -->
		<slot></slot>
	</li>
	...


	<script>
		...
		props: ['list']
	</script>

```
